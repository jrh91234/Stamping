รองรับ: GET /records, POST /records

ตัวอย่าง Google Apps Script (Backend)

สร้าง Script ใหม่ แล้วผูกกับ Google Sheet จากนั้น Deploy เป็น Web App (Execute as: Me, Access: Anyone with link)

function doGet(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('records');
  const action = e.parameter.action;

  if (action === 'list') {
    const values = sheet.getDataRange().getValues();
    const headers = values.shift();
    const records = values.map((row) => Object.fromEntries(headers.map((h, i) => [h, row[i]])));
    return ContentService.createTextOutput(JSON.stringify({ records })).setMimeType(ContentService.MimeType.JSON);
  }

  return ContentService.createTextOutput(JSON.stringify({ error: 'Unknown action' })).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('records');
  const payload = JSON.parse(e.postData.contents || '{}');

  if (payload.action === 'create') {
    sheet.appendRow([
      payload.date,
      payload.shift,
      payload.line,
      payload.machine,
      payload.plannedMinutes,
      payload.downtimeMinutes,
      payload.totalCount,
      payload.goodCount,
      payload.idealCycleTime,
      payload.notes || ''
    ]);

    return ContentService.createTextOutput(JSON.stringify({ ok: true })).setMimeType(ContentService.MimeType.JSON);
  }

  return ContentService.createTextOutput(JSON.stringify({ error: 'Unknown action' })).setMimeType(ContentService.MimeType.JSON);
}
